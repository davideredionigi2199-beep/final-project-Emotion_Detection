"""
Questo modulo implementa un server Flask per l'analisi delle emozioni.
Utilizza il pacchetto EmotionDetection per processare il testo e 
restituire i punteggi emotivi e l'emozione dominante.
"""
from flask import Flask, render_template, request
from EmotionDetection.emotion_detection import emotion_detector

# Inizializza l'applicazione Flask
app = Flask("Emotion Detector")

@app.route("/emotionDetector")
def sent_analyzer():
    """
    Riceve il testo dalla richiesta web, lo analizza tramite emotion_detector
    e restituisce una stringa formattata con i risultati o un messaggio di errore.
    """
    # Recupera il testo da analizzare dagli argomenti della query
    text_to_analyze = request.args.get('textToAnalyze')

    # Chiama la funzione di rilevamento delle emozioni
    response = emotion_detector(text_to_analyze)

    # Estrae i punteggi e l'emozione dominante
    anger = response.get('anger')
    disgust = response.get('disgust')
    fear = response.get('fear')
    joy = response.get('joy')
    sadness = response.get('sadness')
    dominant_emotion = response.get('dominant_emotion')

    # Gestione del caso in cui l'input sia nullo o non valido
    if dominant_emotion is None:
        return "Invalid text! Please try again!."

    # Restituisce la risposta formattata come richiesto dal cliente
    return (
        f"For the given statement, the system response is 'anger': {anger}, "
        f"'disgust': {disgust}, 'fear': {fear}, 'joy': {joy} and 'sadness': {sadness}. "
        f"The dominant emotion is {dominant_emotion}."
    )

@app.route("/")
def render_index_page():
    """
    Rende la pagina principale (index.html) dell'applicazione.
    """
    return render_template('index.html')

if __name__ == "__main__":
    # Avvia il server sulla porta 5000
    app.run(host="0.0.0.0", port=5000)
